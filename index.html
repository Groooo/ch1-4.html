#include<stdio.h>
#include<Windows.h>
#include<conio.h>
#define system_BOTTOM_Y 12
#define console_BOTTOM_Y 20
#define console_private_X 45

void SetConsoleView()
  {
      system("mode con:cols=100 lines=25");
      system("title Hacking DarkWeb.");
  }

void GotoXY(int x, int y)
  {
      COORD Pos;
      Pos.X = 2 * x;
      Pos.Y = y;
      SetConsoleCursorPosition(GetStdHandle(STD_OUTPUT_HANDLE), Pos);
  }

int GetKeyDown()
  {
      if (_kbhit() != 0)
          {
              return _getch();
          }
      return 0;
  }

void Drawsystem(int systemY)
  {
      GotoXY(0, systemY);
      static bool legFlag = true;
      if (legFlag)
          {
              legFlag = false;
          }
      else
          {
              printf("What is Essential is Invisible to the Eye.");
              legFlag = true;
          }
  }

void Drawconsole(int consoleX)
  {
      GotoXY(consoleX, console_BOTTOM_Y);
      printf("twinkle");
      GotoXY(consoleX, console_BOTTOM_Y + 1);
      printf("sparkle");
      GotoXY(consoleX, console_BOTTOM_Y + 2);
      printf("glitter");
  }

int main(void)
  {
      SetConsoleView();
      bool ishacking = false;
      bool isBottom = true;
      const int block = 3;

      int systemY = system_BOTTOM_Y;
      int consoleX = console_BOTTOM_X;

      while (true)
          {
              if (GetKeyDown() == 'z' && isBottom)
                  {
                      ishacking = true;
                      isBottom = false;
                  }
              if (ishacking)
                  {
                      systemY -= block;
                  }
              else
                  {
                      systemY += block;
                  }
              if (systemY >= system_BOTTOM_Y)
                  {
                      systemY = system_BOTTOM_Y;
                      isBottom = true;
                  }
              if (systemY <= 3)
                  {
                      ishacking = false;
                  }
              consoleX -= 2;
              if (consoleX <= 0)
                  {
                      consoleX = console_BOTTOM_X;
                  }

              Drawsystem(systemY);
              Drawconsole(consoleX);
              Sleep(80);
              system("cls");
          }
      printf("When you get stuck, start over");
      system("pause");
      return 0;
  }
